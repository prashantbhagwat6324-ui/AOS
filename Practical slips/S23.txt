    <script>
        // Function to rotate images
        function rotateImages(direction) {
            const images = document.querySelectorAll('.image-container img');
            images.forEach(img => {
                if (direction === 'clockwise') {
                    img.style.transform = 'rotate(90deg)';
                } else if (direction === 'anticlockwise') {
                    img.style.transform = 'rotate(-90deg)';
                }
            });
        }

        // Neo4j Graph Model and Cypher Queries for Automobile Industry

        const createNodes = `
        // Create Vehicle Types
        CREATE (:Vehicle {type: "Two-Wheeler", category: "Motorcycle", price: 50000}),
               (:Vehicle {type: "Four-Wheeler", category: "Car", price: 500000}),
               (:Vehicle {type: "Two-Wheeler", category: "Scooter", price: 30000}),
               (:Vehicle {type: "Four-Wheeler", category: "SUV", price: 800000}),
               (:Vehicle {type: "Four-Wheeler", category: "Truck", price: 1000000});

        // Create Customers
        CREATE (:Customer {name: "John Doe", purchase_date: "2023-11-01"}),
               (:Customer {name: "Jane Smith", purchase_date: "2023-11-02"}),
               (:Customer {name: "Emily Johnson", purchase_date: "2023-11-05"});

        // Create Relationships
        MATCH (v:Vehicle {type: "Two-Wheeler"}), (c:Customer {name: "John Doe"})
        CREATE (c)-[:PURCHASED]->(v);

        MATCH (v:Vehicle {type: "Four-Wheeler"}), (c:Customer {name: "Jane Smith"})
        CREATE (c)-[:PURCHASED]->(v);

        MATCH (v:Vehicle {type: "Two-Wheeler"}), (c:Customer {name: "Emily Johnson"})
        CREATE (c)-[:PURCHASED]->(v);

        MATCH (v:Vehicle {type: "Four-Wheeler"}), (c:Customer {name: "Emily Johnson"})
        CREATE (c)-[:PURCHASED]->(v);

        MATCH (c:Customer {name: "Emily Johnson"}), (v:Vehicle {type: "Two-Wheeler"})
        CREATE (c)-[:RECOMMENDED]->(v);
        `;

        const queries = {
            listFourWheelerCharacteristics: `
            MATCH (v:Vehicle {type: "Four-Wheeler"})
            RETURN v.category AS Category, v.price AS Price;
            `,
            customersWithTwoWheeler: `
            MATCH (c:Customer)-[:PURCHASED]->(v:Vehicle {type: "Two-Wheeler"})
            RETURN c.name AS CustomerName;
            `,
            customersWithMultipleVehicles: `
            MATCH (c:Customer)-[:PURCHASED]->(v:Vehicle)
            WITH c, COUNT(v) AS vehicleCount
            WHERE vehicleCount > 1
            RETURN c.name AS CustomerName;
            `,
            mostRecommendedVehicle: `
            MATCH (c:Customer)-[:RECOMMENDED]->(v:Vehicle)
            RETURN v.type AS VehicleType, COUNT(c) AS Recommendations
            ORDER BY Recommendations DESC
            LIMIT 1;
            `,
        };

        console.log('Nodes Creation:');
        console.log(createNodes);

        console.log('Queries:');
        console.log('a) List of characteristics of four-wheeler types:', queries.listFourWheelerCharacteristics);
        console.log('b) Customers who bought a two-wheeler vehicle:', queries.customersWithTwoWheeler);
        console.log('c) Customers who bought more than one type of vehicle:', queries.customersWithMultipleVehicles);
        console.log('d) Most recommended vehicle type:', queries.mostRecommendedVehicle);
    </script>
